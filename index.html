import requests
from PIL import Image
import numpy as np

# Testing instances
image = Image.open(file='image.jpg' )

# Instantiate model
url = 'mlblocks.com/predict/'
options = {
api_reference: 'API_ACCESS_TOKEN',
data: image
}

# Get predictions response from trained model
predictions = request.get(url, data=options)
